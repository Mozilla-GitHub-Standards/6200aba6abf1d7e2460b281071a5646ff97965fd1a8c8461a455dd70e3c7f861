# This change is required when you move your auth configuration (for server-reg # or server-core) from auth.sql to user.sql


create table users_new (
userid int not null primary key,
username varbinary(32) not null unique key,
password varbinary(128),
accountStatus int default 1,
mail varchar(64),
mailVerified int default 0,
syncNode varchar(64)
);

insert into users_new select id, username, password_hash, status, email, 0, null from users;

rename table users to users_old;
rename table users_new to users;
